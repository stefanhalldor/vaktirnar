Also cache in-flight promises per coordinate to prevent duplicate concurrent requests for the same point:

```ts
const inflight = new Map<string, Promise<string | null>>()

export async function resolvePlaceLabel(lat: number, lon: number): Promise<string | null> {
  if (process.env.NEXT_PUBLIC_ENABLE_REVERSE_GEOCODE !== 'true') return null

  const key = cacheKey(lat, lon)
  if (cache.has(key)) return cache.get(key)!
  if (inflight.has(key)) return inflight.get(key)!

  const promise = (async () => {
    try {
      const url = `https://nominatim.openstreetmap.org/reverse?lat=${lat}&lon=${lon}&format=json&zoom=10&accept-language=is`
      const res = await fetch(url)
      if (!res.ok) { cache.set(key, null); return null }
      const data = await res.json()
      const addr = data.address ?? {}
      const name =
        addr.village ?? addr.hamlet ?? addr.town ?? addr.suburb ??
        addr.city ?? addr.municipality ?? addr.county ?? data.name ?? null
      cache.set(key, name)
      return name
    } catch {
      cache.set(key, null)
      return null
    } finally {
      inflight.delete(key)
    }
  })()

  inflight.set(key, promise)
  return promise
}
```

  let cancelled = false

  if (summary.isOrigin) {
    setPlaceLabel(originName)
    setPlaceLoading(false)
    return
  }

    setPlaceLabel(destinationName)

## Step 2 — Gust display rule

In `PointDetailsPanel`, the weather values block:

**Before:**
```tsx
{(summary.windMs > 0 || summary.gustMs > 0 || summary.precipMmPerHour > 0) && (
  <p>
    {tf('metricWind')}: {summary.windMs.toFixed(1)} m/s ·{' '}
    {tf('metricGust')}: {summary.gustMs.toFixed(1)} m/s ·{' '}
    {tf('metricPrecip')}: {summary.precipMmPerHour.toFixed(1)} mm/klst
  </p>
)}
```

**After:** show `Hviður` only when `gustMs > windMs`, show `Úrkoma` only when `precipMmPerHour > 0`:

```tsx
{(summary.windMs > 0 || summary.precipMmPerHour > 0) && (
  <p>
    {tf('metricWind')}: {summary.windMs.toFixed(1)} m/s
    {summary.gustMs > summary.windMs && (
      <>{' '}· {tf('metricGust')}: {summary.gustMs.toFixed(1)} m/s
        {/* Non-color visual cue when gusts are the decisive metric */}
        {summary.status !== 'graent' && summary.gustMs > summary.windMs && (
          <span className="ml-1 text-[10px] font-medium opacity-70">↑</span>
        )}
      </>
    )}
    {summary.precipMmPerHour > 0 && (
      <>{' '}· {tf('metricPrecip')}: {summary.precipMmPerHour.toFixed(1)} mm/klst</>
    )}
  </p>
)}
```

The `↑` non-color indicator when gusts are above wind and the point is non-green signals "gusts are elevated here" without relying on color.
